.model small
.stack 100h
.data
    msg1 db 'Enter a character: $'
    msg2 db 'Character is not a letter (A-Z).$'

.code
main proc
    mov ax, @data
    mov ds, ax

    ; Prompt user 
    mov ah, 09h
    lea dx, msg1
    int 21h

    ; Read character
    mov ah, 01h
    int 21h        ; Character in AL
    mov bl, al     ; Save input to BL for later use

    ; New line
    mov ah, 02h
    mov dl, 0Dh
    int 21h
    mov dl, 0Ah
    int 21h

    ; Check if it's between 'A' and 'Z'
    cmp bl, 'A'
    jl not_letter
    cmp bl, 'Z'
    jg not_letter

    ; Valid capital letter â€” print it
    mov ah, 02h
    mov dl, bl
    int 21h
    jmp end

not_letter:
    mov ah, 09h
    lea dx, msg2
    int 21h

end:
    mov ah, 4Ch
    int 21h 
    
    
 main endp
end main
