.MODEL SMALL
.STACK 100H
.DATA
I DB 'ENTER INPUT: $'
O DB 10,13, 'OUTPUT: $'
.CODE
INPUT PROC
MOV AX,@DATA
MOV DS,AX
MOV AH,9
LEA DX,I
INT 21H
MOV AH,1
INT 21H
MOV BL,AL
RET
INPUT ENDP
OUTPUT PROC
MOV AX,@DATA
MOV DS,AX
MOV AH,9
LEA DX,O
INT 21H
MOV AH,2
MOV DL,BL
INT 21H
RET
OUTPUT ENDP
 
MAIN PROC
MOV AX,@DATA
MOV DS,AX
CALL INPUT
CALL OUTPUT
MOV AH,4CH
INT 21H
MAIN ENDP
END MAIN
